## 群友的DNS配置
```
dns:
  enable: true
  ipv6: true
  use-hosts: false
  prefer-h3: false
  cache-algorithm: arc
  enhanced-mode: fake-ip
  default-nameserver:
    - quic://223.5.5.5#DIRECT
    - quic://223.6.6.6#DIRECT
  nameserver:
    - quic://dns.alidns.com#DIRECT
  proxy-server-nameserver:
    - quic://dns.alidns.com
  fallback:
    - https://dns.google/dns-query#PROXY&h3=false # 指定策略组和使用 HTTP/3
  nameserver-policy:
    "+.openwrt.ai":
      - quic://dns.alidns.com#DIRECT
    "+.msftconnecttest.com,+.msftncsi.com,+.bing.com,+.googleapis.cn":
      - https://dns.google/dns-query#PROXY&h3=false # 指定策略组和使用 HTTP/3
  fallback-filter:
    geoip: false
    geoip-code: CN
    ipcidr:
      - 0.0.0.0/8
      - 10.0.0.0/8
      - 100.64.0.0/10
      - 127.0.0.0/8
      - 169.254.0.0/16
      - 172.16.0.0/12
      - 192.0.0.0/24
      - 192.0.2.0/24
      - 192.88.99.0/24
      - 192.168.0.0/16
      - 198.18.0.0/15
      - 198.51.100.0/24
      - 203.0.113.0/24
      - 224.0.0.0/4
      - 240.0.0.0/4
      - 255.255.255.255/32
  fake-ip-range: 198.18.0.1/16
  fake-ip-filter:
    - "*.lan"
    - "*.local"
    - localhost.ptlogin2.qq.com
 ```
